$application_html5 = $row->application_html5;
$end = $row->end;

if ($row->conditional == '' || $row->conditional === null) {
    $begin_conditional = '';
    $end_conditional   = '';
} else {
    if ((int)$application_html5 == 1) {
        $end = '>';
    } else {
        $end = '/>';
    }
    $begin_conditional = '<!--[' . $row->conditional . ' ]>';
    $end_conditional   = '<![endif]-->' . chr(10);
}
?>
<? echo $begin_conditional; ?>
    <link rel="stylesheet"
    href="<? echo $row->url; ?>"<? if ((int)$row->application_html5 == 0): ?>
    type="<? echo $row->mimetype; ?>"<? endif; ?><? if ($row->media != null): ?>
    media="<? echo $row->media; ?>"<? endif; ?><? if (trim(
        $row->attributes
    ) != ''
): ?><? echo $row->attributes; ?><? endif; ?><? echo $end; ?><? echo $end_conditional;
